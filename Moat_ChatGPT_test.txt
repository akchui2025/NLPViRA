// ==================== TYPES (PRESERVED) ====================

interface VersionInfo {
    currentVersion: number
    history: any[]
}

interface Message {
    id: string
    user?: string
    bot?: string
    processed?: boolean
    role: "user" | "assistant" | "system"
    requestId?: string
    timestamp?: number
    source?: string
    actions?: any[]
}

interface MoatChatGPTProps {
    tuning: Record<string, any>
    scores: {
        industryScore: { pre: number; post: number }
        executiveScore: { pre: number; post: number }
        archetypeScore: { pre: number; post: number }
        careerArcScore: { pre: number; post: number }
        behavioralScore: { pre: number; post: number }
    }
    apexIndex: number
    fileContents: { [filename: string]: NormalizedItem[] }
    versionInfo?: VersionInfo
    style?: React.CSSProperties
    messages: Message[]
    setMessages: React.Dispatch<React.SetStateAction<Message[]>>
    onLoad?: () => void
    onMouseEnter?: (e: React.MouseEvent) => void
    onMouseLeave?: (e: React.MouseEvent) => void
    setOpenSections?: (sections: { [idx: number]: boolean }) => void
    setShowLeftPanel?: (show: boolean) => void
    setShowRightPanel?: (show: boolean) => void
    setChatVisible?: (visible: boolean) => void
    setActivePanel?: (panel: string) => void
    setActiveRightPanel?: (panel: string) => void
    setIsPanelExtended?: (extended: boolean) => void
    setIsChatPopupOpen?: (open: boolean) => void
    cursorEngine: ViraCursorEngine | null
}

// ==================== EditCommand Interface (PRESERVED) ====================
interface EditCommand {
    method:
        | "navigateToSection"
        | "replaceAll"
        | "replace"
        | "insert"
        | "delete"
        | "multiReplace"
        | "rewriteSection"
        | "deleteSection"
    section?: string
    findText?: string
    replaceText?: string
    content?: string
    edits?: Array<{ target: string; content: string }>
}

// ==================== MAIN COMPONENT ====================
const MoatChatGPT: React.FC<MoatChatGPTProps> = ({
    tuning,
    scores,
    apexIndex,
    fileContents,
    versionInfo,
    style,
    messages,
    setMessages,
    onLoad,
    onMouseEnter,
    onMouseLeave,
    setOpenSections,
    setShowLeftPanel,
    setShowRightPanel,
    setChatVisible,
    setActivePanel,
    setActiveRightPanel,
    setIsPanelExtended,
    setIsChatPopupOpen,
    cursorEngine,
}) => {
    // ==================== STATE (PRESERVED) ====================
    const messageIdCounter = useRef(0)
    const generateMessageId = () =>
        `msg-${Date.now()}-${messageIdCounter.current++}`
    const [userInput, setUserInput] = useState("")
    const [isRequestActive, setIsRequestActive] = useState(false)
    const [isButtonHovered, setIsButtonHovered] = useState(false)
    const [activeFilename, setActiveFilename] = useState<string>("")
    const [viraFramerBridge, setViraFramerBridge] =
        useState<ViRAFramerBridgeIntegration | null>(null)
    const [parsedResume, setParsedResume] = useState<{
        parsedItems: NormalizedItem[]
    } | null>(null)
    const [documentEditor, setDocumentEditor] =
        useState<ViRADocumentEditor | null>(null)

    const messagesEndRef = useRef<HTMLDivElement>(null)
    const abortControllerRef = useRef<AbortController | null>(null)
    const processedMessageIds = useRef<string[]>([])

    // ==================== HOOKS (PRESERVED) ====================
    const documentBridge = useDocumentBridge()
    const { getDocumentJSON, getActiveDocuments, updateDocument } =
        documentBridge

    const resumeContextResult = useResumeDataContext()
    const resumeContext = resumeContextResult || {
        setParsedResume: () => {},
        file: null,
        parsedResume: null,
        saveParsedResume: () => {},
        hasUnsavedChanges: false,
        setHasUnsavedChanges: () => {},
        setLoading: () => {},
    }

    const { file } = resumeContext

    // ==================== INITIALIZATION (PRESERVED) ====================
    useEffect(() => {
        const handleVersionUpdated = (data: {
            filename: string
            version: number
        }) => {
            if (data.filename === activeFilename) {
                console.log(
                    `[MoatChatGPT] Version updated to ${data.version} for ${data.filename}`
                )
                setParsedResume((prev) =>
                    prev ? { ...prev, _version: data.version } : null
                )
            }
        }
        EventBus.on("VersionUpdated", handleVersionUpdated)
        return () => EventBus.off("VersionUpdated", handleVersionUpdated)
    }, [activeFilename, setParsedResume])

    useEffect(() => {
        let engine
        try {
            engine = new ViRAFramerBridgeIntegration()
            setViraFramerBridge(engine)
        } catch (e) {
            console.error(
                "[MoatChatGPT] ViRAFramerBridgeIntegration initialization failed:",
                e
            )
            setViraFramerBridge(null)
        }
        return () => {}
    }, [])

    useEffect(() => {
        if (file?.name && !documentEditor) {
            const editor = ViRADocumentEditor.getInstance(documentBridge)
            const initialized = editor.initialize(file.name)
            if (initialized) {
                setDocumentEditor(editor)
                console.log("[MoatChatGPT] Document editor initialized")
            }
        }
    }, [file?.name, documentEditor, documentBridge])

    // ==================== CORE FUNCTIONS (DEFINITIVE FIX) ====================
    const extractSectionText = (
        items: NormalizedItem[],
        sectionName: string
    ): string | null => {
        const sectionNameLower = sectionName.toLowerCase().trim()
        const startIndex = items.findIndex(
            (item) =>
                item.type === "header" &&
                item.text.toLowerCase().trim().includes(sectionNameLower)
        )

        if (startIndex === -1) return null

        let endIndex = startIndex + 1
        while (endIndex < items.length && items[endIndex].type !== "header") {
            endIndex++
        }

        // Get the header and all its content items
        const sectionItems = items.slice(startIndex, endIndex)
        // Join their text content, preserving line breaks
        return sectionItems.map((item) => item.text).join("\n")
    }

    const sendMessageToApi = useCallback(
        async (
            text: string,
            currentMessages: Message[]
        ): Promise<string | null> => {
            console.log(`[MoatChatGPT] Processing text: "${text}"`)
            abortControllerRef.current = new AbortController()
            const signal = abortControllerRef.current.signal

            try {
                const activeDocs = getActiveDocuments()
                if (!activeDocs || activeDocs.length === 0)
                    return "No active document available."

                const filename = activeDocs[0]
                const documentContext = "" // Context will be built on the fly
                let original_text_to_rewrite = null

                // Detect if this is a rewrite request
                const rewriteMatch = text.match(
                    /rewrite the '([^']*)' section/i
                )
                if (rewriteMatch && rewriteMatch[1]) {
                    const sectionName = rewriteMatch[1]
                    const allItems = getDocumentJSON(filename)
                    if (allItems) {
                        original_text_to_rewrite = extractSectionText(
                            allItems,
                            sectionName
                        )
                        if (!original_text_to_rewrite) {
                            return `I couldn't find a section named '${sectionName}'. Please check the name and try again.`
                        }
                    }
                }

                const apiMessages = currentMessages.slice(-6).map((msg) => ({
                    role: msg.role as "user" | "assistant",
                    content: msg.user || msg.bot || "",
                }))
                apiMessages.push({ role: "user", content: text })

                const prompt = VIRA_PROMPT({
                    context: getDocumentJSON(filename)
                        .map((i) => i.text)
                        .join("\n"),
                    userInput: text,
                    tuning,
                    scores,
                    apexIndex,
                    versionInfo,
                    original_text_to_rewrite, // Pass the extracted text to the prompt
                })

                const response = await fetch(
                    "https://nextjs-boilerplate-vira.vercel.app/api/chat",
                    {
                        method: "POST",
                        headers: { "Content-Type": "application/json" },
                        body: JSON.stringify({
                            model: "gpt-4o",
                            messages: [
                                { role: "system", content: prompt },
                                ...apiMessages,
                            ],
                        }),
                        signal,
                    }
                )

                if (signal.aborted) throw new Error("Request aborted")
                if (!response.ok)
                    throw new Error(`API Error: ${response.statusText}`)

                const data = await response.json()
                const responseText = data?.choices?.[0]?.message?.content || ""

                let parsedResponse
                try {
                    const jsonMatch = responseText.match(/\{[\s\S]*\}/)
                    if (jsonMatch && jsonMatch[0]) {
                        parsedResponse = JSON.parse(jsonMatch[0])
                    } else {
                        parsedResponse = { response: responseText, actions: [] }
                    }
                } catch (parseError) {
                    console.error(
                        "[Parser] Failed to parse JSON. Raw:",
                        responseText,
                        parseError
                    )
                    return (
                        responseText ||
                        `I'm sorry, I received an invalid response.`
                    )
                }

                const conversationalResponse =
                    parsedResponse.response || "I've processed your request."
                const uiActions = parsedResponse.actions || []

                if (uiActions.length > 0) {
                    console.log(
                        "[Flow] Emitting events for actions:",
                        uiActions
                    )
                    for (const action of uiActions) {
                        const payload = action.payload || {}
                        const command: EditCommand = {
                            method: action.action as EditCommand["method"],
                            findText: payload.find_text,
                            replaceText: payload.replace_text,
                        }
                        EventBus.emit("ViRACommand", command)
                    }
                    return conversationalResponse
                }

                return conversationalResponse
            } catch (error) {
                console.error("[MoatChatGPT] sendMessageToApi Error:", error)
                return `Sorry, an error occurred: ${(error as Error).message}`
            } finally {
                abortControllerRef.current = null
            }
        },
        [
            getActiveDocuments,
            getDocumentJSON,
            tuning,
            scores,
            apexIndex,
            versionInfo,
        ]
    )

    const handleSend = useCallback(async () => {
        if (!userInput.trim() || isRequestActive) return

        const input = userInput.trim()
        const newUserMessage: Message = {
            id: generateMessageId(),
            user: input,
            processed: true,
            role: "user",
        }
        const updatedMessages = [...messages, newUserMessage]

        setMessages(updatedMessages)
        setUserInput("")
        setIsRequestActive(true)

        try {
            const responseMessage = await sendMessageToApi(
                input,
                updatedMessages
            )
            if (responseMessage) {
                setMessages((prev) => [
                    ...prev,
                    {
                        id: generateMessageId(),
                        bot: responseMessage,
                        processed: true,
                        role: "assistant",
                    },
                ])
            }
        } catch (error: unknown) {
            console.error("[MoatChatGPT] Send error:", error)
            setMessages((prev) => [
                ...prev,
                {
                    id: generateMessageId(),
                    bot: `Error: ${(error as Error).message}`,
                    processed: true,
                    role: "assistant",
                },
            ])
        } finally {
            setIsRequestActive(false)
        }
    }, [userInput, isRequestActive, messages, setMessages, sendMessageToApi])

    const handleSendExternal = useCallback(
        async (text: string) => {
            if (!text.trim() || isRequestActive) return

            const trimmedText = text.trim()
            const requestId = `ext-${Date.now()}`
            const newUserMessage: Message = {
                id: generateMessageId(),
                user: `🎤 ${trimmedText}`,
                processed: true,
                timestamp: Date.now(),
                source: "external",
                requestId,
                role: "user",
            }

            const updatedMessages = [...messages, newUserMessage]
            setMessages(updatedMessages)
            setIsRequestActive(true)

            try {
                const responseMessage = await sendMessageToApi(
                    trimmedText,
                    updatedMessages
                )
                if (responseMessage) {
                    setMessages((prev) => [
                        ...prev,
                        {
                            id: generateMessageId(),
                            bot: responseMessage,
                            processed: true,
                            timestamp: Date.now(),
                            source: "vira",
                            requestId,
                            role: "assistant",
                        },
                    ])
                }
            } catch (error: unknown) {
                console.error("[MoatChatGPT] Send external error:", error)
                const errorMessage =
                    error instanceof Error ? error.message : "Unknown error"
                setMessages((prev) => [
                    ...prev,
                    {
                        id: generateMessageId(),
                        bot: `Error processing soundbite: ${errorMessage}`,
                        processed: true,
                        timestamp: Date.now(),
                        source: "error",
                        requestId,
                        role: "assistant",
                    },
                ])
            } finally {
                setIsRequestActive(false)
            }
        },
        [isRequestActive, messages, setMessages, sendMessageToApi]
    )

    const handleStop = useCallback(() => {
        if (abortControllerRef.current) {
            console.log("[MoatChatGPT] Aborting current request")
            abortControllerRef.current.abort()
            abortControllerRef.current = null
        }
        setIsRequestActive(false)
        setMessages((prev) => [
            ...prev,
            {
                id: generateMessageId(),
                bot: "Request cancelled by user.",
                processed: true,
                role: "assistant",
            },
        ])
    }, [setMessages])

    useEffect(() => {
        if (onLoad) {
            onLoad()
        }
    }, [onLoad])

    useEffect(() => {
        const styleTag = document.createElement("style")
        styleTag.innerHTML = GLOBAL_STYLES
        document.head.appendChild(styleTag)

        return () => {
            if (document.head.contains(styleTag)) {
                document.head.removeChild(styleTag)
            }
        }
    }, [])

    useEffect(() => {
        if (messagesEndRef.current) {
            messagesEndRef.current.scrollIntoView({
                behavior: "smooth",
                block: "end",
            })
        }
    }, [messages])

    useEffect(() => {
        return () => {
            if (abortControllerRef.current) {
                abortControllerRef.current.abort()
            }
        }
    }, [])

    useEffect(() => {
        return () => {
            processedMessageIds.current = []
        }
    }, [])

    useEffect(() => {
        const handleExternalMessage = (data: any) => {
            if (isRequestActive) {
                return
            }

            let messageText = ""
            if (typeof data === "string") {
                messageText = data
            } else if (data && typeof data === "object") {
                messageText =
                    data.soundbite ||
                    data.text ||
                    data.message ||
                    data.content ||
                    data.transcript ||
                    ""
            }

            if (
                messageText &&
                typeof messageText === "string" &&
                messageText.trim()
            ) {
                handleSendExternal(messageText.trim())
            }
        }

        ;(window as any).onSoundbiteTrigger = handleExternalMessage

        return () => {
            ;(window as any).onSoundbiteTrigger = null
        }
    }, [handleSendExternal, isRequestActive])

    const handleConfirmMessage = useCallback(
        (message: string) => {
            setMessages((prev) => [
                ...prev,
                {
                    id: generateMessageId(),
                    bot: message,
                    processed: true,
                    timestamp: Date.now(),
                    source: "vira",
                    role: "assistant",
                },
            ])
        },
        [setMessages]
    )

    return (
        <div
            className="moat-chat-container"
            contentEditable={false}
            spellCheck={false}
            style={{
                display: "flex",
                flexDirection: "column",
                height: "100%",
                width: "100%",
                background: "black",
                border: "1px solid #111111",
                borderRadius: "12px",
                ...style,
            }}
            onMouseEnter={onMouseEnter}
            onMouseLeave={onMouseLeave}
        >
            <div
                style={{
                    padding: "12px",
                    color: "#ffffff",
                    display: "flex",
                    alignItems: "center",
                    justifyContent: "space-between",
                    height: "50px",
                    borderBottom: "1px solid #111111",
                    flexShrink: 0,
                }}
            >
                <div style={{ display: "flex", alignItems: "center" }}>
                    <img
                        src="https://archive.org/download/moat_logo/moat_logo.png"
                        alt="Moat Logo"
                        style={{ height: "32px", marginRight: "12px" }}
                    />
                    <span style={{ fontSize: "19px", fontWeight: "600" }}>
                        moät.chat
                    </span>
                </div>
            </div>
            <div
                className="moat-chat-messages"
                contentEditable={false}
                spellCheck={false}
                style={{
                    flex: 1,
                    padding: "12px",
                    height: "calc(100% - 137px)",
                    overflowY: "auto",
                }}
            >
                {messages.map((msg) => (
                    <MessageComponent key={msg.id} msg={msg} />
                ))}
                {isRequestActive && (
                    <div
                        style={{
                            color: "#00FFC2",
                            textAlign: "center",
                            padding: "12px",
                        }}
                    >
                        Processing request...{" "}
                        <span className="editing-indicator" />
                    </div>
                )}
                <div ref={messagesEndRef} style={{ height: "1px" }} />
            </div>
            <div
                style={{
                    height: "87px",
                    padding: "12px",
                    background: "black",
                    borderTop: "1px solid #111111",
                    position: "relative",
                    bottom: "0",
                    zIndex: 2,
                }}
            >
                <div
                    style={{
                        display: "flex",
                        alignItems: "center",
                        backgroundColor: "#252525",
                        borderRadius: "24px",
                        padding: "4px 4px 4px 16px",
                    }}
                >
                    <textarea
                        value={userInput}
                        onChange={(e) => setUserInput(e.target.value)}
                        placeholder="Message your agent..."
                        disabled={isRequestActive}
                        spellCheck={false}
                        style={{
                            flex: 1,
                            minHeight: "44px",
                            maxHeight: "120px",
                            border: "none",
                            background: "transparent",
                            color: "#FFFFFF",
                            fontSize: "15px",
                            outline: "none",
                            padding: "10px 0",
                            resize: "none",
                        }}
                        onKeyDown={(e) => {
                            if (e.key === "Enter" && !e.shiftKey) {
                                e.preventDefault()
                                handleSend()
                            }
                        }}
                    />
                    <button
                        onClick={isRequestActive ? handleStop : handleSend}
                        disabled={!userInput.trim() && !isRequestActive}
                        onMouseEnter={() => setIsButtonHovered(true)}
                        onMouseLeave={() => setIsButtonHovered(false)}
                        style={{
                            backgroundColor: "transparent",
                            border: `1px solid ${
                                isRequestActive ? "#F511BC" : "#00FFC2"
                            }`,
                            borderRadius: "50%",
                            width: "36px",
                            height: "36px",
                            display: "flex",
                            alignItems: "center",
                            justifyContent: "center",
                            cursor: "pointer",
                            marginLeft: "8px",
                            opacity:
                                !userInput.trim() && !isRequestActive ? 0.5 : 1,
                        }}
                    >
                        {isRequestActive ? (
                            <StopIcon isHovered={isButtonHovered} />
                        ) : (
                            <PaperPlaneIcon isHovered={isButtonHovered} />
                        )}
                    </button>
                </div>
            </div>
        </div>
    )
}

export default React.memo(MoatChatGPT)
